datasource db { provider = "postgresql"; url = env("DATABASE_URL") }
generator client { provider = "prisma-client-js" }

model Org { id String @id @default(cuid()); name String; users User[]; projects Project[]; leads Lead[]; createdAt DateTime @default(now()) }

enum UserRole { ADMIN MANAGER AGENT }

model User {
  id String @id @default(cuid())
  orgId String
  email String @unique
  name String
  role UserRole @default(AGENT)
  passwordHash String
  createdAt DateTime @default(now())
  org Org @relation(fields:[orgId], references:[id])
  activities Activity[]
}

model Project {
  id String @id @default(cuid())
  orgId String
  name String
  location String?
  builder String?
  createdAt DateTime @default(now())
  org Org @relation(fields:[orgId], references:[id])
  leads Lead[]
}

enum Stage { NEW CONTACTED SCHEDULED VISITED NEGOTIATION BOOKED DROPPED }

model Lead {
  id String @id @default(cuid())
  orgId String
  projectId String?
  name String?
  phone String
  email String?
  source String      // google_ads | meta | website | csv
  campaign String?
  adset String?
  ad String?
  gclid String?
  fbclid String?
  utmSource String?
  utmMedium String?
  utmCampaign String?
  bedroomsPref Int?
  budgetMin Int?
  budgetMax Int?
  locationPref String?
  stage Stage @default(NEW)
  score Int @default(0)
  assignedToId String?
  duplicateOf String?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  org Org @relation(fields:[orgId], references:[id])
  project Project? @relation(fields:[projectId], references:[id])
  assignedTo User? @relation(fields:[assignedToId], references:[id])
  activities Activity[]
  siteVisits SiteVisit[]
}

enum ActivityType { CALL WHATSAPP EMAIL NOTE STATUS_CHANGE ASSIGNMENT }

model Activity {
  id String @id @default(cuid())
  leadId String
  userId String?
  type ActivityType
  note String?
  payload Json?
  at DateTime @default(now())
  lead Lead @relation(fields:[leadId], references:[id])
  user User? @relation(fields:[userId], references:[id])

enum VisitStatus { SCHEDULED DONE NO_SHOW RESCHEDULED }

model SiteVisit {
  id String @id @default(cuid())
  leadId String
  projectId String?

  status VisitStatus @default(SCHEDULED)

  createdAt DateTime @default(now())

export function scoreLead(l: Partial<Lead>): number {
  let s = 0;
  if ([2,3].includes(l.bedroomsPref ?? 0)) s += 20;
  if ((l.budgetMax ?? 0) >= 8000000) s += 15;           // ≥ ₹80L
  if ((l.locationPref ?? "").toLowerCase().includes("whitefield")) s += 10;
  if ((l.source ?? "") === "google_ads") s += 15;
  if (l.duplicateOf) s -= 10;
  return s;
}

# 6) API Routes (Next.js App Router)
Implement with Zod validation and proper error handling:
- POST `/api/leads`                         → create: normalize, dedupe, score, assign (round-robin among AGENTs), record Activity
- GET  `/api/leads`                         → list with filters: `stage`, `assignedTo`, `source`, `q` (name/phone/email/campaign), `from/to`; server pagination; mask PII in list
- PATCH `/api/leads/[id]`                   → update fields, stage, assignment; record Activity (STATUS_CHANGE/ASSIGNMENT)
- POST `/api/leads/[id]/activities`         → add NOTE/CALL/WHATSAPP/EMAIL
- POST `/api/leads/[id]/schedule-visit`     → create SiteVisit + Activity; set stage SCHEDULED
- POST `/api/hooks/meta`                    → normalize Meta payload; ingest + Activity
- POST `/api/hooks/google-ads`              → normalize Google Ads Lead Form payload; ingest + Activity
- GET  `/api/reports/source-metrics`        → totals + **CPL, CPSV, CPB** grouped by `source/campaign` within date range
- GET  `/api/reports/funnel`                → counts per stage in date range
- GET  `/api/admin/offline-conversions`     → list JSON payloads prepared on stage=BOOKED when `gclid` exists (preview only)
- GET  `/api/cron/sla`                      → mark NEW leads older than 15 min as overdue; send assignment nudges
- GET  `/api/cron/reminders`                → send T‑24h and T‑2h site‑visit reminders to agent (and optional lead)

Notes:
- All writes create an `Activity` row.
- For `/api/leads` search, use Postgres text search across name/phone/email/campaign; keep it efficient with indexes.

# 9) Cron Automation with Cloudflare Workers

Sample Worker (JS):
export default {
  async scheduled(controller, env, ctx) {
    const base = env.BASE_URL; // e.g., https://your-app.vercel.app
    await fetch(`${base}/api/cron/sla`);
    await fetch(`${base}/api/cron/reminders`);
  }
}

`wrangler.toml`:
name = "crm-cron"
main = "src/index.js"
compatibility_date = "2024-07-01"
[triggers]
crons = ["*/5 * * * *"]  # every 5 minutes
